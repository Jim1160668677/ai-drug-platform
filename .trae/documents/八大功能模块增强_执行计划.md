```python
def add_noise(gradients, noise_multiplier, max_norm):
    # 裁剪梯度
    total_norm = torch.norm(torch.stack([g.norm() for g in gradients]))
    clip_coef = min(1.0, max_norm / (total_norm + 1e-6))
    clipped = [g * clip_coef for g in gradients]
    # 添加高斯噪声
    noise = torch.normal(0, noise_multiplier * max_norm, size=g.shape)
    return [c + noise for c, n in zip(clipped, noise)]
```

```python
async def _step4_generate_hypotheses(self, project_id, targets, molecules, treatments):
    """Step 4: 假设生成 — 基于前 3 步结果自动生成研究假设"""
    from app.services.knowledge.hypothesis_generator import HypothesisGenerator
    from app.services.llm.orchestrator import LLMOrchestrator
    from app.services.llm.client import get_llm_client_with_config

    llm_client, llm_config = await get_llm_client_with_config(self.db)
    llm_orchestrator = LLMOrchestrator(self.db, llm_client, llm_config=llm_config)
    generator = HypothesisGenerator(self.db)

    # 收集上下文
    context = await self._collect_pipeline_context(project_id, targets, molecules, treatments)

    # 规则 + LLM 混合生成
    hypotheses = await generator.generate_hybrid(context, llm_orchestrator)

    return {"status": "success", "hypotheses_generated": len(hypotheses), "hypotheses": hypotheses}
```

修改 `run()` 方法：在 Step 3 之后增加 Step 4 调用。
