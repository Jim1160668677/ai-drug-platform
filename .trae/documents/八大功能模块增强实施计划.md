async def _compute_affinity_real(self, smiles: str, target_pdb: str) -> float:
    """使用 DiffDock 进行真实分子对接"""
    try:
        import httpx
        from app.core.config import settings
        if not settings.NVIDIA_NIM_API_KEY:
            raise ValueError("NVIDIA_NIM_API_KEY 未配置")

        async with httpx.AsyncClient(timeout=120) as client:
            response = await client.post(
                settings.DIFFDOCK_NIM_URL,
                headers={"Authorization": f"Bearer {settings.NVIDIA_NIM_API_KEY}"},
                json={"ligand_smiles": smiles, "protein_pdb": target_pdb},
            )
            result = response.json()
            # DiffDock 返回置信度分数（0-1，越高越好）
            return float(result.get("confidence", 0.5))

#### Step 5.2：差分隐私增强

**修改文件**：`backend/app/services/optimizer/federated_learning.py`

```python
def _apply_differential_privacy(
    self,
    weights: Dict[str, Any],
    epsilon: float = 1.0,
    delta: float = 1e-5,
) -> Dict[str, Any]:
    """应用差分隐私（DP-SGD）"""
    # 在权重上添加高斯噪声
    import numpy as np
    sigma = np.sqrt(2 * np.log(1.25 / delta)) / epsilon
    noisy_weights = {}
    for layer, w in weights.items():
        if isinstance(w, (int, float)):
            noise = np.random.normal(0, sigma * abs(w) * 0.01)
            noisy_weights[layer] = w + noise
        else:
            noisy_weights[layer] = w
    return noisy_weights
```
