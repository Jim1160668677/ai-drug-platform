```python
"""生信数据多格式导入导出"""
import csv
import io
import json
from typing import Any, Dict, List
from xml.etree import ElementTree as ET


class BioDataIO:
    """生信数据导入导出工具"""

    @staticmethod
    async def import_csv(content: bytes, has_header: bool = True) -> Dict[str, Any]:
        """CSV 导入 — 基因表达矩阵
        格式：第一列为基因名，后续列为样本表达值
        """
        text = content.decode("utf-8")
        reader = csv.reader(io.StringIO(text))
        rows = list(reader)
        if not rows:
            return {"genes": {}, "samples": []}
        header = rows[0] if has_header else [f"sample_{i}" for i in range(len(rows[0]) - 1)]
        samples = header[1:]
        genes = {}
        for row in rows[1:] if has_header else rows:
            if len(row) < 2:
                continue
            gene = row[0]
            values = [float(v) if v else 0.0 for v in row[1:]]
            genes[gene] = values
        return {"genes": genes, "samples": samples, "format": "csv"}

    @staticmethod
    async def import_json(content: bytes) -> Dict[str, Any]:
        """JSON 导入 — {genes: {gene: [values]}, samples: [...]}"""
        data = json.loads(content.decode("utf-8"))
        return {"genes": data.get("genes", {}), "samples": data.get("samples", []), "format": "json"}

    @staticmethod
    async def import_fasta(content: bytes) -> Dict[str, Any]:
        """FASTA 导入 — 序列数据
        返回 {sequences: [{id, seq, length}], format: "fasta"}
        """
        text = content.decode("utf-8")
        sequences = []
        current_id = None
        current_seq = []
        for line in text.splitlines():
            line = line.strip()
            if line.startswith(">"):
                if current_id is not None:
                    sequences.append({"id": current_id, "seq": "".join(current_seq), "length": len(current_seq)})
                current_id = line[1:].split()[0] if len(line) > 1 else f"seq_{len(sequences)}"
                current_seq = []
            elif line:
                current_seq.append(line)
        if current_id is not None:
            sequences.append({"id": current_id, "seq": "".join(current_seq), "length": len(current_seq)})
        return {"sequences": sequences, "count": len(sequences), "format": "fasta"}

    @staticmethod
    async def import_vcf(content: bytes) -> Dict[str, Any]:
        """VCF 导入 — 变异信息（简化版）
        返回 {variants: [{chrom, pos, id, ref, alt}], format: "vcf"}
        """
        text = content.decode("utf-8")
        variants = []
        for line in text.splitlines():
            if line.startswith("#") or not line.strip():
                continue
            fields = line.split("\t")
            if len(fields) < 5:
                continue
            variants.append({
                "chrom": fields[0], "pos": int(fields[1]) if fields[1].isdigit() else 0,
                "id": fields[2], "ref": fields[3], "alt": fields[4],
            })
        return {"variants": variants, "count": len(variants), "format": "vcf"}

    @staticmethod
    async def import_mtx(content: bytes) -> Dict[str, Any]:
        """MTX 导入 — 稀疏矩阵（Market Matrix 格式简化版）"""
        text = content.decode("utf-8")
        lines = [l for l in text.splitlines() if l.strip() and not l.startswith("%")]
        if not lines:
            return {"matrix": [], "format": "mtx"}
        # 第一行：rows cols nnz
        header = lines[0].split()
        rows, cols, nnz = int(header[0]), int(header[1]), int(header[2])
        matrix = []
        for line in lines[1:1 + nnz]:
            parts = line.split()
            if len(parts) >= 3:
                matrix.append({"row": int(parts[0]), "col": int(parts[1]), "value": float(parts[2])})
        return {"matrix": matrix, "shape": [rows, cols], "nnz": nnz, "format": "mtx"}

    @staticmethod
    async def export_csv(data: Dict[str, Any]) -> bytes:
        """CSV 导出 — 差异表达/聚类结果
        data: {genes: [...]} 或 {clusters: [...]}
        """
        output = io.StringIO()
        writer = csv.writer(output)
        if "genes" in data:
            writer.writerow(["gene", "log2fc", "pvalue", "padj", "regulation", "significant"])
            for g in data["genes"]:
                writer.writerow([g.get("gene"), g.get("log2fc"), g.get("pvalue"),
                                 g.get("padj"), g.get("regulation"), g.get("significant")])
        elif "clusters" in data:
            writer.writerow(["gene", "cluster_id", "pca_x", "pca_y"])
            for c in data["clusters"]:
                writer.writerow([c.get("gene"), c.get("cluster_id"), c.get("pca_x"), c.get("pca_y")])
        elif "pathways" in data:
            writer.writerow(["id", "name", "pvalue", "ratio", "genes"])
            for p in data["pathways"]:
                writer.writerow([p.get("id"), p.get("name"), p.get("pvalue"),
                                 p.get("ratio"), ";".join(p.get("genes", []))])
        else:
            writer.writerow(["key", "value"])
            for k, v in data.items():
                writer.writerow([k, str(v)[:100]])
        return output.getvalue().encode("utf-8")

    @staticmethod
    async def export_json(data: Dict[str, Any]) -> bytes:
        """JSON 导出 — 任意分析结果"""
        return json.dumps(data, ensure_ascii=False, indent=2, default=str).encode("utf-8")

    @staticmethod
    async def detect_format(filename: str, content: bytes) -> str:
        """自动检测文件格式"""
        ext = filename.rsplit(".", 1)[-1].lower() if "." in filename else ""
        if ext in ("csv", "tsv", "json", "fasta", "fa", "vcf", "mtx"):
            return "tsv" if ext == "tsv" else ext
        # 内容嗅探
        head = content[:200].decode("utf-8", errors="ignore").strip()
        if head.startswith(">"):
            return "fasta"
        if head.startswith("##fileformat=VCF"):
            return "vcf"
        if head.startswith("{"):
            return "json"
        return "csv"


__all__ = ["BioDataIO"]
```

```python
async def import_data_file(
    project_id: UUID = Query(...),
    name: str = Query(...),
    file: UploadFile = File(...),
    db: AsyncSession = Depends(get_db),
    current_user: User = Depends(get_current_user),
):
    """导入多格式数据文件（CSV/JSON/FASTA/VCF/MTX）— 解析后创建 Dataset 记录"""
    from app.services.analyzer.data_io import BioDataIO
    content = await file.read()
    fmt = await BioDataIO.detect_format(file.filename or "", content)
    if fmt == "csv":
        parsed = await BioDataIO.import_csv(content)
    elif fmt == "tsv":
        parsed = await BioDataIO.import_csv(content)  # TSV 复用 CSV
    elif fmt == "json":
        parsed = await BioDataIO.import_json(content)
    elif fmt in ("fasta", "fa"):
        parsed = await BioDataIO.import_fasta(content)
    elif fmt == "vcf":
        parsed = await BioDataIO.import_vcf(content)
    elif fmt == "mtx":
        parsed = await BioDataIO.import_mtx(content)
    else:
        raise ValidationError(f"不支持的格式: {fmt}")
    # 创建 Dataset 记录
    data_type_map = {"csv": "rna_seq", "tsv": "rna_seq", "json": "rna_seq",
                     "fasta": "fasta", "vcf": "wes", "mtx": "scrna_seq"}
    dataset = Dataset(
        project_id=project_id, name=name,
        data_type=data_type_map.get(fmt, "other"),
        source=file.filename, storage_path="",
        file_size=len(content), file_format=fmt,
        parse_status=ParseStatus.COMPLETED,
        parsed_summary=parsed, uploaded_by=current_user.id,
    )
    db.add(dataset)
    await db.flush()
    return success_response({
        "id": str(dataset.id), "name": name, "format": fmt,
        "parsed": parsed, "dataset": DatasetResponse.model_validate(dataset).model_dump(),
    })
```

```python
    self,
    target_ids: list,
    seed_smiles: str | None = None,
    n: int = 10,
    constraints: dict | None = None,

    """多靶点协同设计

    1. 为每个靶点独立生成候选分子
    2. 计算每个分子对所有靶点的结合亲和力（Mock 模式下用随机评分）
    3. 按多靶点综合评分（均值+平衡性）排序
    Returns:
        {molecules: [{smiles, structure_url, properties, binding_affinities, multi_target_score}],
         model_info: {strategy, targets, n_requested}}
    """
    import random
    random.seed(42)

    if not target_ids:
        return {"molecules": [], "model_info": {"error": "no targets"}}

    # 1. 收集所有靶点
    targets = []
    for tid in target_ids:
        t = await self.db.get(Target, tid) if hasattr(self, 'db') else None
        targets.append({"id": tid, "name": getattr(t, "name", tid) if t else tid})

    # 2. 为第一个靶点生成候选分子（其他靶点用于评分）
    primary = target_ids[0]
    gen_result = await self.generate_molecules(
        target_id=primary, strategy="fragment", n=n * 2, seed_smiles=seed_smiles, constraints=constraints or {})
    candidates = gen_result.get("molecules", [])

    # 3. 评估每个候选
    results = []
    for mol in candidates[:n * 2]:
        smiles = mol.get("smiles", "")
        if not smiles:
            continue
        props = assess_druglikeness(smiles)
        if props.get("error"):
            continue
        # 多靶点亲和力：Mock 模式随机生成 -12 ~ -6 kcal/mol
        binding_affinities = {}
        for t in targets:
            binding_affinities[t["id"]] = round(random.uniform(-12.0, -6.0), 2)
        # 综合评分：平均亲和力（越负越好）归一化到 0-1
        avg_affinity = sum(binding_affinities.values()) / len(binding_affinities)
        # 平衡性：标准差越小越好
        if len(binding_affinities) > 1:
            variance = sum((v - avg_affinity) ** 2 for v in binding_affinities.values()) / len(binding_affinities)
            balance = 1.0 / (1.0 + variance)
        else:
            balance = 1.0
        # 归一化亲和力（-12 → 1.0, -6 → 0.0）
        affinity_score = max(0.0, min(1.0, (-avg_affinity - 6.0) / 6.0))
        drug_score = props.get("druglikeness_score", 0) / 100.0
        multi_target_score = round(0.5 * affinity_score + 0.3 * drug_score + 0.2 * balance, 4)

        results.append({
            "smiles": smiles,
            "structure_url": f"https://cactus.nci.nih.gov/chemical/structure/{smiles}/image",
            "properties": {
                "mw": props.get("mw"),
                "logp": props.get("logp"),
                "hbd": props.get("hbd"),
                "hba": props.get("hba"),
                "passes_rule_of_five": props.get("passes_rule_of_five"),
                "druglikeness_score": props.get("druglikeness_score"),
            },
            "binding_affinities": binding_affinities,
            "multi_target_score": multi_target_score,
        })

    # 4. 按综合评分降序
    results.sort(key=lambda x: x["multi_target_score"], reverse=True)
    return {
        "molecules": results[:n],
        "model_info": {
            "strategy": "multi_target_synergy",
            "targets": targets,
            "n_requested": n,
            "n_returned": len(results[:n]),
        },
    }
```

```python
"""假设自动生成引擎 — 规则推理 + LLM 增强"""
import logging
import random
from typing import Any, Dict, List

from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

logger = logging.getLogger(__name__)


class HypothesisGenerator:
    """从前期分析结果自动生成研究假设"""

    def __init__(self, db: AsyncSession):
        self.db = db

    async def generate(self, project_id: str, context: Dict[str, Any] | None = None) -> List[Dict[str, Any]]:
        """生成假设主流程
        
        Args:
            project_id: 项目 ID
            context: 前序分析结果（可选，缺省时自动从 DB 拉取）
        Returns:
            [{name, description, mechanism, evidence, validation_methods, confidence, source_job_id}]
        """
        # 1. 收集前序分析结果
        ctx = context or await self._collect_context(project_id)
        # 2. 规则推理
        rule_based = self._rule_inference(ctx)
        # 3. LLM 增强（Mock 模式直接使用规则结果）
        enhanced = await self._llm_enhance(rule_based, ctx)
        # 4. 结构化
        return self._structure(enhanced, ctx)

    async def _collect_context(self, project_id: str) -> Dict[str, Any]:
        """收集项目所有分析结果"""
        from app.models.target import Target
        from app.models.molecule import Molecule
        from app.models.hypothesis import Hypothesis
        from app.models.dataset import Dataset

        ctx = {"project_id": project_id, "targets": [], "molecules": [], "datasets": []}

        # 靶点
        target_stmt = select(Target).where(Target.project_id == project_id).limit(20)
        targets = (await self.db.execute(target_stmt)).scalars().all()
        ctx["targets"] = [{"id": str(t.id), "name": t.name, "gene_symbol": getattr(t, "gene_symbol", None),
                           "confidence_score": getattr(t, "confidence_score", 0)} for t in targets]

        # 分子
        mol_stmt = select(Molecule).limit(20)
        mols = (await self.db.execute(mol_stmt)).scalars().all()
        ctx["molecules"] = [{"id": str(m.id), "smiles": m.smiles, "logp": getattr(m, "logp", None),
                             "mw": getattr(m, "molecular_weight", None)} for m in mols]

        # 数据集
        ds_stmt = select(Dataset).where(Dataset.project_id == project_id).limit(10)
        datasets = (await self.db.execute(ds_stmt)).scalars().all()
        ctx["datasets"] = [{"id": str(d.id), "name": d.name, "summary": d.parsed_summary} for d in datasets]

        return ctx

    def _rule_inference(self, ctx: Dict[str, Any]) -> List[Dict[str, Any]]:
        """规则推理 — 基于靶点、分子、通路生成假设"""
        hypotheses = []
        targets = ctx.get("targets", [])
        molecules = ctx.get("molecules", [])

        # 规则 1：高置信度靶点 → 假设该靶点是关键药物靶点
        for t in targets[:3]:
            if (t.get("confidence_score") or 0) > 0.7:
                hypotheses.append({
                    "rule": "high_confidence_target",
                    "name": f"靶点 {t['name']} 是 {ctx['project_id']} 项目的关键药物靶点",
                    "evidence": [f"置信度评分 {t['confidence_score']:.2f}"],
                    "mechanism": f"通过调控 {t['name']} 表达可影响疾病进程",
                })

        # 规则 2：多靶点协同假设
        if len(targets) >= 2:
            names = "、".join(t["name"] for t in targets[:3])
            hypotheses.append({
                "rule": "multi_target_synergy",
                "name": f"同时调控 {names} 可产生协同治疗效果",
                "evidence": [f"项目内共识别 {len(targets)} 个候选靶点"],
                "mechanism": "多靶点协同调控可同时干预多个疾病通路",
            })

        # 规则 3：类药分子 → 候选药物假设
        for m in molecules[:3]:
            logp = m.get("logp")
            if logp is not None and 0 <= logp <= 5:
                hypotheses.append({
                    "rule": "druglike_molecule",
                    "name": f"分子 {m['smiles'][:20]}... 具有良好类药性，可作为候选药物",
                    "evidence": [f"logP={logp}（Lipinski 范围内）", f"MW={m.get('mw')}"],
                    "mechanism": "该分子符合 Lipinski 五规则，具备良好口服生物利用度",
                })

        return hypotheses

    async def _llm_enhance(self, rule_based: List[Dict], ctx: Dict) -> List[Dict]:
        """LLM 增强 — 将规则结果传入 LLM 生成自然语言描述（Mock 模式跳过）"""
        from app.core.config import settings
        if settings.USE_MOCK or not rule_based:
            return rule_based
        try:
            from app.services.llm.router import LLMRouter
            router = LLMRouter(db=self.db)
            prompt = self._build_llm_prompt(rule_based, ctx)
            response = await router.complete(prompt, tier="fast_screen")
            # 简单解析 LLM 输出
            for i, h in enumerate(rule_based):
                h["llm_description"] = response.get("content", "")[:500]
            return rule_based
        except Exception as e:
            logger.warning(f"LLM 增强降级: {e}")
            return rule_based

    @staticmethod
    def _build_llm_prompt(rule_based: List[Dict], ctx: Dict) -> str:
        return (
            f"基于以下规则推理结果，生成科学严谨的研究假设描述：\n"
            f"项目上下文：{len(ctx.get('targets', []))} 个靶点，{len(ctx.get('molecules', []))} 个分子\n"
            f"规则假设：{rule_based}\n"
            f"请为每个假设生成更专业的描述和预期验证方法。"
        )

    def _structure(self, hypotheses: List[Dict], ctx: Dict) -> List[Dict[str, Any]]:
        """结构化为最终输出"""
        random.seed(42)
        result = []
        for h in hypotheses:
            result.append({
                "name": h.get("name", ""),
                "description": h.get("llm_description") or h.get("name", ""),
                "mechanism": h.get("mechanism", ""),
                "strategy": h.get("rule", "rule_based"),
                "evidence": h.get("evidence", []),
                "validation_methods": self._suggest_validation(h),
                "confidence": round(random.uniform(0.6, 0.9), 2),
                "source_job_id": ctx.get("project_id"),
                "analysis_config": {"rule": h.get("rule")},
            })
        return result

    @staticmethod
    def _suggest_validation(h: Dict) -> List[str]:
        """根据假设类型建议验证方法"""
        rule = h.get("rule", "")
        if rule == "high_confidence_target":
            return ["体外敲低实验验证靶点功能", "CRISPR-Cas9 基因编辑验证", "临床样本表达量分析"]
        elif rule == "multi_target_synergy":
            return ["双靶点联合用药实验", "Bliss 独立性模型分析协同效应", "动物模型验证"]
        elif rule == "druglike_molecule":
            return ["体外 ADMET 实验", "细胞活性测试", "药代动力学实验"]
        return ["文献检索验证", "实验验证"]


__all__ = ["HypothesisGenerator"]
```

```python
async def add_dp_noise(self, weights: dict, epsilon: float = 1.0) -> dict:
    """差分隐私 — 拉普拉斯噪声注入权重"""
    import random
    return {k: v + random.laplace(0, max(abs(v), 0.01) / epsilon) for k, v in weights.items()}

async def secure_aggregate(self, client_weights: list) -> dict:
    """安全聚合 — 加法掩码"""
    if not client_weights:
        return {}
    # 简化实现：取均值（实际应使用秘密共享+掩码）
    keys = set()
    for w in client_weights:
        keys.update(w.keys())
    result = {}
    for k in keys:
        vals = [w.get(k, 0) for w in client_weights]
        result[k] = sum(vals) / len(vals)
    return result
```

```python
async def run_with_config(self, config: dict, current_user=None) -> dict:
    """按用户配置运行流水线
    
    config: {steps: [str], inject_hypothesis: bool, project_id, ...}
    """
    results = {}
    hypothesis_tracking = []
    steps = config.get("steps", ["data_analysis", "target_discovery", "molecule_design", "hypothesis_generation", "treatment"])

    for step in steps:
        try:
            if step == "data_analysis":
                results["data_analysis"] = await self._step_data_analysis(config)
            elif step == "target_discovery":
                results["targets"] = await self._step_target_discovery(config, results)
            elif step == "molecule_design":
                targets = results.get("targets", [])
                if config.get("multi_target") and len(targets) > 1:
                    results["molecules"] = await self._step_multi_target_design(targets, config)
                else:
                    results["molecules"] = await self._step_molecule_design(targets, config)
            elif step == "hypothesis_generation":
                hyps = await self._step_generate_hypotheses(config, results)
                results["hypotheses"] = hyps
                hypothesis_tracking.append({"step": "hypothesis_generation", "generated": len(hyps)})
                # 若启用假设注入，将假设作为后续步骤输入
                if config.get("inject_hypothesis") and hyps:
                    results["_active_hypothesis"] = hyps[0]
            elif step == "treatment":
                results["treatments"] = await self._step_treatment(config, results)
            hypothesis_tracking.append({"step": step, "status": "completed"})
        except Exception as e:
            hypothesis_tracking.append({"step": step, "status": "failed", "error": str(e)})

    results["hypothesis_tracking"] = hypothesis_tracking
    return results

async def _step_generate_hypotheses(self, config, results) -> list:
    """假设生成步骤"""
    from app.services.analyzer.hypothesis_generator import HypothesisGenerator
    generator = HypothesisGenerator(self.db)
    context = {
        "targets": results.get("targets", []),
        "molecules": results.get("molecules", {}).get("molecules", []) if isinstance(results.get("molecules"), dict) else results.get("molecules", []),
    }
    return await generator.generate(config["project_id"], context)

async def _step_multi_target_design(self, targets, config) -> dict:
    """多靶点分子设计步骤"""
    from app.services.analyzer.molecule_designer import MoleculeDesigner
    designer = MoleculeDesigner(self.db)
    target_ids = [t.get("id") or t.get("target_id") for t in targets if t.get("id") or t.get("target_id")]
    return await designer.design_multi_target(target_ids, n=config.get("molecules_per_target", 10))
```

## 假设与决策

1. **ERR_ABORTED** — 不做代码修复，确认所有页面存在且返回 200，属于 Next.js 14 开发模式瞬时行为。
2. **Mock 模式优先** — 所有新功能必须在 `USE_MOCK=true` 下可用（用户当前环境）。
3. **模型注册同步** — 每个新模型（ClinicalFeedback、FederatedJob、FederatedNode、PipelineConfig、PipelineRun）必须在 `models/__init__.py`、`db/session.py`、`tests/conftest.py` 三处注册。
4. **外键使用 `String(36)`** — 避免 SQLAlchemy UUID 类型兼容性问题。
5. **前端导入 `api/client`** — 所有新 API 函数复用 `import { api } from './client'`。
6. **数据库迁移** — 新模型通过 `init_db()` 自动 `create_all()`，无需 Alembic 迁移（开发模式）。
7. **clinical_feedback 端点放在 treatments.py** — 更贴近临床应用语义。
8. **功能 4 隐私增强** — 直接修改 `federated_learning.py` 添加方法，不新建 `optimizer/privacy.py`（避免与现有 `services/privacy/` 混淆）。
9. **功能 6 流水线** — 保持现有 `run` 方法不变，新增 `run_with_config` 方法支持自定义步骤。
