```python
async def apply_clinical_feedback(
    self,
    feedback_data: Dict[str, Any],
    treatment_id: str,
) -> Dict[str, Any]:
    """应用临床反馈到闭环

    1. 计算实际vs预期疗效差异
    2. 记录不良反应
    3. 触发方案优化建议
    4. 关联实验数据（通过 treatment_id）
    """
    from app.models.treatment import Treatment
    treatment = await self.db.get(Treatment, treatment_id)

    expected_efficacy = (treatment.efficacy_score if treatment else 0.5) or 0.5
    efficacy_map = {"complete": 1.0, "partial": 0.6, "stable": 0.4, "progressive": 0.1}
    actual_efficacy = efficacy_map.get(feedback_data.get("efficacy", ""), 0.5)

    efficacy_diff = actual_efficacy - expected_efficacy
    adverse_reactions = feedback_data.get("adverse_reactions") or []

    suggestions = []
    if efficacy_diff < -0.2:
        suggestions.append("实际疗效显著低于预期，建议调整用药方案或更换药物")
    elif efficacy_diff < 0:
        suggestions.append("实际疗效略低于预期，建议微调剂量")
    else:
        suggestions.append("实际疗效符合或超过预期，方案有效")

    if adverse_reactions:
        suggestions.append(f"记录到 {len(adverse_reactions)} 项不良反应，建议评估风险收益比")

    return {
        "treatment_id": treatment_id,
        "loop_stage": "clinical_validation",
        "expected_efficacy": round(expected_efficacy, 3),
        "actual_efficacy": round(actual_efficacy, 3),
        "efficacy_diff": round(efficacy_diff, 3),
        "adverse_reactions_count": len(adverse_reactions),
        "optimization_suggestions": suggestions,
        "next_stage": "dry_prediction_update",
    }
```

async def evaluate_job(self, job_id: str) -> Dict[str, Any]:
    """评估联邦学习任务性能"""
    job = self._jobs.get(job_id)
    if not job:
        return {"job_id": job_id, "error": "任务不存在"}
    rounds_history = job.get("rounds_history", [])
    rounds_completed = len([r for r in rounds_history if r.get("aggregated")])
    # 从提交的 metrics 中提取性能指标
    all_metrics = []
    for r in rounds_history:
        for s in r.get("submissions", []):
            if s.get("metrics"):
                all_metrics.append(s["metrics"])
    avg_accuracy = 0.0
    avg_loss = 0.0
    if all_metrics:
        accuracies = [m.get("accuracy", 0) for m in all_metrics if "accuracy" in m]
        losses = [m.get("loss", 0) for m in all_metrics if "loss" in m]
        avg_accuracy = sum(accuracies) / len(accuracies) if accuracies else 0
        avg_loss = sum(losses) / len(losses) if losses else 0
    return {
        "job_id": job_id,
        "rounds_completed": rounds_completed,
        "total_rounds": job.get("num_rounds", 0),
        "status": job.get("status"),
        "metrics": {
            "accuracy": round(avg_accuracy, 4),
            "loss": round(avg_loss, 4),
            "convergence": "converged" if rounds_completed >= job.get("num_rounds", 0) else "in_progress",
        },
        "framework": job.get("framework"),
    }

async def check_compliance(self, job_id: str) -> Dict[str, Any]:
    """合规检查 — 数据隐私保护法规"""
    job = self._jobs.get(job_id)
    if not job:
        return {"job_id": job_id, "error": "任务不存在"}
    config = job.get("config", {})
    checks = []
    # 1. 差分隐私预算
    dp_epsilon = config.get("dp_epsilon")
    checks.append({
        "name": "差分隐私预算 (ε)",
        "passed": dp_epsilon is not None and 0 < dp_epsilon <= 10.0,
        "detail": f"ε={dp_epsilon}" if dp_epsilon else "未配置差分隐私",
    })
    # 2. 数据驻留
    data_residency = config.get("data_residency")
    checks.append({
        "name": "数据驻留合规",
        "passed": data_residency is not None,
        "detail": f"驻留区域: {data_residency}" if data_residency else "未配置数据驻留区域",
    })
    # 3. 客户端知情同意
    consent_obtained = config.get("consent_obtained", False)
    checks.append({
        "name": "客户端知情同意",
        "passed": consent_obtained,
        "detail": "已获取" if consent_obtained else "未获取知情同意",
    })
    # 4. 安全聚合
    secure_agg = config.get("secure_aggregation", False)
    checks.append({
        "name": "安全聚合协议",
        "passed": secure_agg,
        "detail": "已启用" if secure_agg else "未启用安全聚合",
    })
    compliant = all(c["passed"] for c in checks)
    return {
        "job_id": job_id,
        "compliant": compliant,
        "checks": checks,
        "recommendations": [] if compliant else ["请完善隐私保护配置后再启动训练"],
    }

async def list_nodes(self, status: Optional[str] = None) -> List[Dict[str, Any]]:
    """列出联邦学习节点（多中心协作模式）"""
    nodes = []
    for cid, client in self._clients.items():
        node_info = {
            "node_id": cid,
            "endpoint": client.get("endpoint"),
            "status": client.get("status"),
            "organization": client.get("capabilities", {}).get("organization", "未知"),
            "data_center": client.get("capabilities", {}).get("data_center", "未知"),
            "last_heartbeat": client.get("last_heartbeat"),
            "weights_submitted": client.get("weights_submitted", 0),
        }
        nodes.append(node_info)
    if status:
        nodes = [n for n in nodes if n.get("status") == status]
    return nodes

async def register_node(
    self,
    node_id: str,
    organization: str,
    endpoint: str,
    data_center: str,
    compliance_cert: Optional[Dict] = None,
) -> Dict[str, Any]:
    """注册联邦学习节点（机构级别）"""
    self._clients[node_id] = {
        "client_id": node_id,
        "endpoint": endpoint,
        "capabilities": {
            "organization": organization,
            "data_center": data_center,
            "compliance_cert": compliance_cert,
        },
        "status": "active",
        "registered_at": datetime.now(timezone.utc).isoformat(),
        "last_heartbeat": datetime.now(timezone.utc).isoformat(),
        "weights_submitted": 0,
    }
    logger.info(f"FL node registered: {node_id} ({organization})")
    return {"node_id": node_id, "status": "registered", "organization": organization}

async def run_with_config(
    self,
    project_id: str,
    config: Dict[str, Any],
    hypothesis_id: Optional[str] = None,
) -> Dict[str, Any]:
    """带配置的流水线执行 — 支持自定义步骤和假设集成"""
    steps = config.get("steps", ["target_discovery", "molecule_design", "treatment_planning"])
    step_configs = config.get("step_configs", {})
    hypothesis_tracking = []

    # 加载假设（如果提供）
    hypothesis_context = None
    if hypothesis_id:
        from app.models.hypothesis import Hypothesis
        hyp = await self.db.get(Hypothesis, hypothesis_id)
        if hyp:
            hypothesis_context = {
                "id": str(hyp.id), "name": hyp.name,
                "description": hyp.description, "mechanism": hyp.mechanism,
            }
            hypothesis_tracking.append({
                "step": "init", "action": "hypothesis_loaded",
                "hypothesis_id": str(hyp.id), "hypothesis_name": hyp.name,
            })

    results: Dict[str, Any] = {"project_id": project_id}

    for step in steps:
        try:
            if step == "target_discovery":
                results["target_discovery"] = await self._cfg_step_targets(project_id, step_configs.get(step, {}))
            elif step == "molecule_design":
                results["molecule_design"] = await self._cfg_step_molecules(
                    project_id, step_configs.get(step, {}), results.get("target_discovery", {})
                )
            elif step == "treatment_planning":
                results["treatment_planning"] = await self._cfg_step_treatments(
                    project_id, step_configs.get(step, {}), results
                )
            elif step == "hypothesis_generation":
                hyps = await self._cfg_step_hypotheses(project_id, results)
                results["hypothesis_generation"] = hyps
                if config.get("inject_hypothesis") and hyps:
                    hypothesis_context = hyps[0]
                    hypothesis_tracking.append({
                        "step": "hypothesis_generation", "action": "hypothesis_injected",
                        "hypothesis_id": hyps[0].get("id", ""),
                    })
            hypothesis_tracking.append({"step": step, "status": "completed"})
        except Exception as e:
            hypothesis_tracking.append({"step": step, "status": "failed", "error": str(e)})
            logger.warning(f"Pipeline step {step} failed: {e}")

    results["hypothesis_tracking"] = hypothesis_tracking
    results["hypothesis_context"] = hypothesis_context
    return results

async def _cfg_step_targets(self, project_id: str, cfg: Dict) -> Dict:

## 八、假设与决策

1. **ClinicalFeedback 模型放在 treatment.py**：与 Treatment 模型同文件，语义内聚（v2 计划决策）
2. **模型注册只需修改 `__init__.py`**：`db/session.py` 和 `conftest.py` 已通过模块名导入 `treatment`，新增类自动被注册
3. **联邦学习不新建模型**：直接在内存态 `FederatedLearningService` 添加方法，与现有架构一致
4. **流水线不新建模型**：复用现有 `WorkflowRun` 模型存储运行记录，`run_with_config` 是 `run` 的增强版
5. **假设集成方式**：通过 `hypothesis_id` 参数传入，在 `hypothesis_tracking` 数组中记录应用过程
6. **前端 pipeline 页面新建**：`frontend/app/workbench/pipeline/page.tsx`，需在导航中添加入口（如 workbench/page.tsx）
7. **后端需重启**：新增 ClinicalFeedback 模型后需重启 uvicorn 以触发 `create_all()` 建表
