  from app.models import *  # noqa: F401,F403 — 触发所有模型注册

  def upgrade():
      # 按依赖顺序创建所有表

  def downgrade():
      Base.metadata.drop_all(bind=op.get_bind())

#### P2.1 类型定义 + API 适配（2 文件）

**文件 1**（新建）：`frontend/types/api.ts`
```typescript
export interface ResponseMeta {
  request_id: string;
  duration_ms?: number;
}
export interface PagedMeta extends ResponseMeta {
  page: number;
  page_size: number;
  total: number;
  total_pages: number;
}
export interface ApiResponse<T> {
  success: boolean;
  data: T;
  meta: ResponseMeta;
}
export interface PagedResponse<T> {
  success: boolean;
  data: T[];
  meta: PagedMeta;
}
export interface ErrorResponse {
  success: false;
  error: {
    code: string;
    message: string;
    details?: any;
  };
  meta: ResponseMeta;
}
export type ErrorCode =
  | 'VALIDATION_ERROR' | 'UNAUTHORIZED' | 'FORBIDDEN' | 'NOT_FOUND'
  | 'CONFLICT' | 'GUARDRAIL_BLOCKED' | 'RATE_LIMITED' | 'UPSTREAM_ERROR'
  | 'INTERNAL_ERROR';
```

**文件 2**（修改）：`frontend/lib/api.ts`
- 响应拦截器：检测 `response.data.success` 字段
  - 若为信封格式：透传 `meta` 到 `response.meta`，返回 `response.data.data`（解包）
  - 若非信封（如 `/metrics` 文本）：原样返回
- 错误拦截器：解包 `ErrorResponse.error`，Toast 显示 `error.message`，保留 401 跳登录
- **兼容策略**：保留现有 `.then(r => r.data)` 调用 — 因解包后 `r.data` 即业务数据，双模式兼容
- 添加 `meta` 透传：`response.meta = response.data.meta`

**关键代码**：
```typescript
api.interceptors.response.use(
  (response) => {
    const body = response.data;
    // 信封格式检测
    if (body && typeof body === 'object' && 'success' in body && 'meta' in body) {
      (response as any).meta = body.meta;
      response.data = body.data;  // 解包
    }
    return response;
  },
  (error) => {
    if (error.response) {
      const { status, data } = error.response;
      // 错误信封解包
      if (data && typeof data === 'object' && 'error' in data) {
        const errBody = data as ErrorResponse;
        error.message = errBody.error.message;
        error.code = errBody.error.code;
        // Toast 显示（除 401，避免登出闪烁）
        if (status !== 401) {
          toast.error(`${errBody.error.code}: ${errBody.error.message}`);
        }
      }
      if (status === 401 && typeof window !== 'undefined') {
        localStorage.removeItem('ai_drug_token');
        localStorage.removeItem('ai_drug_user');
        if (window.location.pathname !== '/') {
          window.location.href = '/';
        }
      }
    }
    return Promise.reject(error);
  }
);
```

## 4. 风险与缓解

| 风险 | 缓解措施 |
|---|---|
| rdkit-js/3dmol 体积大导致构建慢 | 动态 import + next/dynamic，仅在使用时加载 |
| Alembic 手动迁移可能遗漏表 | 使用 `Base.metadata.create_all` 反射，不手写 Column |
| 前端 api.ts 解包兼容性 | 信封检测基于 `success` + `meta` 字段，非信封响应原样返回 |
| P4 测试工作量过大 | D15 分两批：后端优先保证 ≥80%，前端关键组件覆盖 |
| admin 拆分破坏书签 | 旧 `/admin` 改为导航页，3 秒后自动跳转 `admin/users`（可选） |
| 路径别名导致 OpenAPI 文档重复 | `include_in_schema=False` 隐藏别名路由 |
